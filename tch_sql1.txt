-- FUNCTION: public.sso_spinsci_execute_get_tch_voicemail_template_data(text)-- DROP FUNCTION public.sso_spinsci_execute_get_tch_voicemail_template_data(text);CREATE OR REPLACE FUNCTION public.sso_spinsci_execute_get_tch_voicemail_template_data(	phonenumber text)    RETURNS jsonb    LANGUAGE 'plpgsql'    COST 100    VOLATILE AS $BODY$DECLARE outputjson jsonb;BEGINSelect template_content::jsonb from outbound_config.sso_spinsci_known_media_settings where template_type='TCHVOICEMAIL' and template_name='VOICEMAIL' into outputjson;  Select replace (outputjson::text,'$dphone',phonenumber) into outputjson;  RETURN outputjson; END; $BODY$;ALTER FUNCTION public.sso_spinsci_execute_get_tch_voicemail_template_data(text)    OWNER TO spinsci;GRANT EXECUTE ON FUNCTION public.sso_spinsci_execute_get_tch_voicemail_template_data(text) TO PUBLIC;GRANT EXECUTE ON FUNCTION public.sso_spinsci_execute_get_tch_voicemail_template_data(text) TO spinsci;
